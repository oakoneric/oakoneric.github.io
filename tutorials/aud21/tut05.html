      1. Ackermann-Funktion 
        #include <stdio.h>

int ack(int x, int y){
    int a;
    if ((x == 0) && (y >= 0))
        return y + 1;
    else if ((x > 0) && (y == 0)) 
        return ack(x-1, 1);
    else if ((x > 0) && (y > 0)){
        a = ack(x, y-1);
        return ack(x-1, a);
    }
}


int main() {
    int x = 0, y = 0, a;
    printf("\nAckermannfunktion\n");
    printf("x = ");
    scanf("%d", &x);
    printf("y = ");
    scanf("%d", &y);
    
    a = ack(x,y);
    
    printf("ack(%i,%i)=%i.\n", x, y, a);
    
    return 0;
}
      2. Fakultät \( n! = n * (n-1) * \dots * 2 * 1\) 
        #include <stdio.h>

int main(){
    int n, i, factorial = 1;
    scanf("%d", &n);

    for (i = 1; i <= n; i = i + 1){
        factorial = factorial * i;
    }
    
    printf("n! = %d\n", factorial);

    return 0;
}
      3. Multiplikationstabelle 
        #include <stdio.h>

int main(){
    int i,j,n;
    scanf("%d", &n);
    
    for ( i=1; i<=n ; i=i+1 ) {
        for ( j=1; j<=n ; j=j+1 ) {
            printf("%4d", i*j);
        }
        printf("\n");
    }
    return 0;
}
